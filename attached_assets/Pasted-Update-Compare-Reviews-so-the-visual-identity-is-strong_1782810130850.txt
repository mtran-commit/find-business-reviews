Fallback logic:
Create helper functions:

function getInitials(name) {
  if (!name) return "?";
  return name
    .split(" ")
    .slice(0, 2)
    .map(word => word[0])
    .join("")
    .toUpperCase();
}

function renderLogo(logoUrl, name, className = "row-logo") {
  if (logoUrl) {
    return `
      <div class="${className}">
        <img src="${logoUrl}" alt="${name} logo" />
      </div>
    `;
  }

  return `
    <div class="${className} fallback">
      ${getInitials(name)}
    </div>
  `;
}

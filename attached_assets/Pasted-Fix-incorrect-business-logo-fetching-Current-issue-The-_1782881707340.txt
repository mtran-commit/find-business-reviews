Add logo validation before rendering:

function isValidBusinessLogo(logoUrl, businessName) {
  if (!logoUrl) return false;

  const url = logoUrl.toLowerCase();
  const nameWords = businessName
    .toLowerCase()
    .replace(/[^a-z0-9\s]/g, " ")
    .split(/\s+/)
    .filter(word => word.length > 2);

  const bannedLogoSignals = [
    "wordpress",
    "wp-content",
    "wp-includes",
    "w.org",
    "wix",
    "squarespace",
    "shopify",
    "google",
    "facebook",
    "instagram",
    "tripadvisor",
    "yelp",
    "favicon",
    "apple-touch-icon",
    "default",
    "placeholder",
    "logo512",
    "logo192"
  ];

  if (bannedLogoSignals.some(signal => url.includes(signal))) {
    return false;
  }

  const hasBusinessNameSignal = nameWords.some(word => url.includes(word));

  return hasBusinessNameSignal;
}

Update logo rendering:

function renderBusinessLogo(logoUrl, businessName, className = "logo-overlay") {
  if (isValidBusinessLogo(logoUrl, businessName)) {
    return `
      <div class="${className}">
        <img src="${logoUrl}" alt="${businessName} logo" />
      </div>
    `;
  }

  return `
    <div class="${className} fallback-logo">
      ${getInitials(businessName)}
    </div>
  `;
}

function showReportGenerating(type = "customer") {
  const message = type === "business"
    ? "Our AI is preparing your Business Reputation Report using available review data and competitor signals. This may take a few seconds."
    : "Our AI is analysing the available Google, Yelp and TripAdvisor review data. This may take a few seconds.";

  showModal(`
    <div class="report-loading">
      <div class="spinner"></div>
      <h2>Report Generating</h2>
      <p>${message}</p>
      <span>Please don’t close this window.</span>
    </div>
  `);
}

Update openCustomerReport():

async function openCustomerReport() {
  if (!currentReports?.customerReport) {
    showReportGenerating("customer");

    let attempts = 0;
    const maxAttempts = 20;

    const waitForReport = setInterval(() => {
      attempts++;

      if (currentReports?.customerReport) {
        clearInterval(waitForReport);
        renderCustomerReport();
      }

      if (attempts >= maxAttempts && !currentReports?.customerReport) {
        clearInterval(waitForReport);
        showModal(`
          <h2>Report could not be generated</h2>
          <p>Report could not be generated right now. Please try again.</p>
        `);
      }
    }, 500);

    return;
  }

  renderCustomerReport();
}

Create renderCustomerReport() and move the existing customer report modal HTML into it:

function renderCustomerReport() {
  const report = currentReports.customerReport;

  showModal(`
    <h2>AI Review Sentiment</h2>
    <p class="modal-intro">This free summary helps customers quickly understand the public review sentiment before they buy, book or visit.</p>

    <div class="report-stat-grid">
      <div><strong>${report.trustScore}/100</strong><span>Trust Score</span></div>
      <div><strong>${report.averageRating}/5</strong><span>Average Rating</span></div>
      <div><strong>${report.platformsChecked}</strong><span>Platforms Checked</span></div>
    </div>

    <h3>Review Summary</h3>
    <p>${report.summary}</p>

    <h3>Rating Gap</h3>
    <p>${report.ratingGapWarning}</p>

    <h3>Recommendation</h3>
    <p>${report.recommendation}</p>
  `);
}

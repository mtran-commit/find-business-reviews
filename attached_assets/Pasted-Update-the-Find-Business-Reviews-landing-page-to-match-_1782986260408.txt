10. JavaScript

Add:

function handleLandingSearch() {
  const businessInput = document.getElementById("businessSearchInput");
  const locationInput = document.getElementById("locationSearchInput");

  const businessQuery = businessInput.value.trim();
  const locationQuery = locationInput.value.trim();

  if (!businessQuery) {
    alert("Please enter a business, service or category.");
    businessInput.focus();
    return;
  }

  const finalQuery = locationQuery ? `${businessQuery} ${locationQuery}` : businessQuery;

  // Use the existing search logic.
  // If current app uses searchInput, set it and call searchBusiness().
  const existingSearchInput = document.getElementById("searchInput");
  if (existingSearchInput) {
    existingSearchInput.value = finalQuery;
  }

  if (typeof searchBusiness === "function") {
    searchBusiness(finalQuery);
  } else {
    window.location.href = `/?q=${encodeURIComponent(finalQuery)}`;
  }
}

Add Enter key support:
["businessSearchInput", "locationSearchInput"].forEach(id => {
  const el = document.getElementById(id);
  if (el) {
    el.addEventListener("keydown", event => {
      if (event.key === "Enter") {
        handleLandingSearch();
      }
    });
  }
});
